        document.addEventListener('DOMContentLoaded', function() {
            const mainNavButtons = document.querySelectorAll('nav.flex-col button[data-tab]');
            const mainTabContents = document.querySelectorAll('.tab-content');
            const subTabButtons = document.querySelectorAll('.sub-tab-button');
            const subTabContents = document.querySelectorAll('.sub-tab-content');
            function activateTab(tabId) {
                mainTabContents.forEach(content => content.classList.remove('active'));
                document.getElementById(tabId + '-tab').classList.add('active');
                mainNavButtons.forEach(button => {
                    if (button.dataset.tab === tabId) {
                        button.classList.add('bg-[#233648]', 'text-white', 'border-primary');
                        button.classList.remove('text-text-secondary', 'hover:bg-[#233648]', 'hover:text-white', 'border-transparent', 'hover:border-[#2f455a]');
                        button.querySelector('.material-symbols-outlined').classList.add('text-primary');
                        button.querySelector('.material-symbols-outlined').classList.remove('group-hover:text-white');
                    } else {
                        button.classList.remove('bg-[#233648]', 'text-white', 'border-primary');
                        button.classList.add('text-text-secondary', 'hover:bg-[#233648]', 'hover:text-white', 'border-transparent', 'hover:border-[#2f455a]');
                        button.querySelector('.material-symbols-outlined').classList.remove('text-primary');
                        button.querySelector('.material-symbols-outlined').classList.add('group-hover:text-white');
                    }
                });
            }
            function activateSubTab(subTabId) {
                subTabContents.forEach(content => content.classList.remove('active'));
                document.getElementById(subTabId + '-sub-tab').classList.add('active');
                subTabButtons.forEach(button => {
                    if (button.dataset.subTab === subTabId) {
                        button.classList.add('active');
                    } else {
                        button.classList.remove('active');
                    }
                });
            }
            mainNavButtons.forEach(button => {
                button.addEventListener('click', () => {
                    activateTab(button.dataset.tab);
                });
            });
            subTabButtons.forEach(button => {
                button.addEventListener('click', () => {
                    activateSubTab(button.dataset.subTab);
                });
            });
            // Initial activation
            activateTab('general');
            activateSubTab('profile');
        });
    </script>
